const slides = document.querySelectorAll('.slide-single');
const slider = [];
let step = 0;
let offset = 0;
let width = document.querySelector('.slide').offsetWidth
const myBtnNext = document.querySelector('.slide-btn-next');

for (let i = 0; i < slides.length; i++) {
    slider[i] = slides[i].src
    slides[i].remove()
}

function draw() {
    let img = document.createElement('img');
    img.src = slider[step];
    img.classList.add('slide-single');
    img.style.left = offset * width + 'px';
    document.querySelector('.slide').appendChild(img);

    if (step + 1 === slider.length) {
        step = 0;
    } else step++
    offset = 1;
}

function left() {
    myBtnNext.onclick = null;
    const slides2 = document.querySelectorAll('.slide-single');
    let offset2 = 0;
    for (let i = 0; i < slides2.length; i++) {
        slides2[i].style.left = offset2 * width - width + 'px'
        offset2++;
    }
    setTimeout(() => {
        slides2[0].remove();
        draw();
        myBtnNext.onclick = left
    }, 1000)
}

draw(); draw();

myBtnNext.onclick = left;
